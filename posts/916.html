BiomeInit.java

public class BiomeInit {
    public static final DeferredRegister<Biome> BIOME = new DeferredRegister<>(ForgeRegistries.BIOMES, Terraria.mod_id);

    public static final RegistryObject<Biome> EXAMPLE_BIOME = BIOME.register("example_biome", () ->
            new ExampleBiome(new Biome.Builder()
                    .waterColor(0xFF0157)
                    .depth(0.125f)
                    .surfaceBuilder(SurfaceBuilder.DEFAULT,
                            new SurfaceBuilderConfig(
                                    Blocks.END_STONE.getDefaultState(),
                                    Blocks.END_STONE.getDefaultState(),
                                    Blocks.SLIME_BLOCK.getDefaultState()))
                    .category(Biome.Category.PLAINS)
                    .downfall(0.1f)
                    .temperature(0.3f)
                    .precipitation(Biome.RainType.RAIN)
                    .waterFogColor(0x114514)
                    .scale(0.5f)
                    .parent(null)));

    public static void registerBiomes(){
        registerBiome(EXAMPLE_BIOME.get(), Type.PLAINS);
        BiomeManager.addBiome(BiomeManager.BiomeType.COOL, new BiomeManager.BiomeEntry(EXAMPLE_BIOME.get(), 100));
    }

    private static void registerBiome(Biome biome, Type... types){
        BiomeDictionary.addTypes(biome, types);
        BiomeManager.addSpawnBiome(biome);
    }
}
