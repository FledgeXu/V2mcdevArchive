@Mod("xiang_plugin")
public class XiangPlugin
{
    private static final Logger LOGGER = LogManager.getLogger();
    private static final Marker MARKER = MarkerManager.getMarker("Xiang's plugin");

    public XiangPlugin() {
        LOGGER.info(MARKER,"The plugin of Xiang is loading.");
        Items.register.register(FMLJavaModLoadingContext.get().getModEventBus());
    }
}
